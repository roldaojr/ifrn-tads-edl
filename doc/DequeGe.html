def popBack
  if @size == 0
    raise "DequeUnderow"
  end
  node = @back
  node.prevNode.nextNode = nil
  @back = node.prevNode
  @size -= 1
  if size < 2
    @front = @back
  end
  return node.element
end

def pushBack(item)
  if self.isFull?
    raise "DequeOverow"
  end
  node = DequeNode.new(item, nil, @back)
  if(@back)
    @back.nextNode = node
  end
  @back = node
  @size += 1
  if size < 2
    @front = @back
  end
end
